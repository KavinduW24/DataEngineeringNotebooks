from pyspark.sql import SparkSession, functions as f 


spark = SparkSession.builder.appName("DataCleaner").getOrCreate()

data = [
    (1, "Alice", 25, "NY"),
    (2, "Bob", None, "LA"),
    (2, "Bob", None, "LA"),
    (3, None, 40, "NY")
]

columns = ["id", "name", "age", "city"]

df = spark.createDataFrame(data, schema=columns)

noDupes = df.distinct()

averageAge = noDupes.agg(f.avg(noDupes.age)).first()[0]



filledAges = noDupes.fillna(averageAge)


filledNames = filledAges.fillna("Unknown")

final = filledNames.filter(f.col("age") >30)

final.show()

spark.stop()

from pyspark.sql import SparkSession, functions as f
from pyspark.sql.window import Window

spark = SparkSession.builder.appName("window func challenge").getOrCreate()

data = [
    ("A", "2024-01-01", 10),
    ("A", "2024-01-02", 20),
    ("A", "2024-01-03", 15),
]

df = spark.createDataFrame(data, ["user", "date", "amount"])

df = df.withColumn("date", f.to_date("date"))

window_spec = Window.partitionBy("user").orderBy("date")

running_window = window_spec.rowsBetween(
    Window.unboundedPreceding,
    Window.currentRow
)

df = df.withColumn(
    "running_total",
    f.sum("amount").over(running_window)
)

df = df.withColumn(
    "previous_amount",
    f.lag("amount").over(window_spec)
)

df = df.withColumn(
    "difference",
    f.col("amount") - f.col("previous_amount")
)

rolling_window = window_spec.rowsBetween(-6, 0)

df = df.withColumn(
    "rolling_avg_7_days",
    f.avg("amount").over(rolling_window)
)

df.show()

spark.stop()

from pyspark.sql import SparkSession, functions as f
from pyspark.sql.window import Window

spark = SparkSession.builder.appName("sessions").getOrCreate()

data = [
    ("Alice", "2024-01-01 09:00:00"),
    ("Alice", "2024-01-01 09:10:00"),
    ("Alice", "2024-01-01 10:00:00"),
    ("Alice", "2024-01-01 10:20:00"),
    ("Bob",   "2024-01-01 09:00:00"),
    ("Bob",   "2024-01-01 09:40:00"),
    ("Bob",   "2024-01-01 09:50:00")
]

df = spark.createDataFrame(data, ["user", "timestamp"])

df = df.withColumn("timestamp", f.to_timestamp("timestamp"))

window_spec = Window.partitionBy("user").orderBy("timestamp")

df = df.withColumn(
    "prev_timestamp",
    f.lag("timestamp").over(window_spec)
)

df = df.withColumn(
    "gap_minutes",
    (f.unix_timestamp("timestamp") - f.unix_timestamp("prev_timestamp")) / 60
)

df = df.withColumn(
    "new_session",
    f.when(
        (df.prev_timestamp.isNull()) | (df.gap_minutes > 30),
        1
    ).otherwise(0)
)

df = df.withColumn(
    "session_id",
    f.sum("new_session").over(
        window_spec.rowsBetween(Window.unboundedPreceding, Window.currentRow)
    )
)

df.select(
    "user",
    "timestamp",
    "gap_minutes",
    "session_id"
).show(truncate=False)


spark.stop()

from pyspark.sql.window import Window

spark = SparkSession.builder.getOrCreate()


orders = spark.read.csv("")

customers = spark.read.csv("")

orders = orders.na.drop(subset=["customer_id"])

orders = orders.fillna({"amount": 0})

orders = orders.dropDuplicates()

customers = customers.dropDuplicates()

joined = orders.join(
    customers,
    "customer_id",
    "left"
)

sales_city = joined.groupBy("city").agg(
    f.sum("amount").alias("total_sales")
)

customer_spend = joined.groupBy("customer_id").agg(
    sum("amount").alias("spending")
)

window = Window.orderBy(customer_spend["spending"].desc())

top3 = (
    customer_spend
    .withColumn("rank", f.row_number().over(window))
    .filter("rank <= 3")
)
